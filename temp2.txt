- csr: hstatus  
  description: >  
    Hypervisor status register. Kiểm soát trạng thái virtualization ở HS-mode.  
  address: 0x600  
  privilege_mode: H  
  rv32:  
    - field_name: SPV  
      description: >  
        Previous virtualisation mode (Saved V bit)  
      type: WARL  
      reset_val: 0  
      msb: 0  
      lsb: 0  
    - field_name: VIEN  
      description: >  
        Virtual Interrupt Enable (for VS-mode)  
      type: WARL  
      reset_val: 0  
      msb: 1  
      lsb: 1  
    # ... (còn các bit khác như SPVP, WPRI hoặc reserved)  
  rv64:  
    - field_name: SPV  
      description: >  
        Previous virtualization mode  
      type: WARL  
      reset_val: 0  
      msb: 0  
      lsb: 0  
    - field_name: VIEN  
      description: >  
        Virtual interrupt enable  
      type: WARL  
      reset_val: 0  
      msb: 1  
      lsb: 1  
    # ... các field khác tùy spec
